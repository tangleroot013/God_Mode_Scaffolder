## The Complete `hatch.py` God-Mode Scaffolder

```python
#!/usr/bin/env python3
"""
hatch.py - AI-Ready Project Bootstrapper
A pure Python infrastructure builder for Claude-optimized projects.
Quack! By Carter the Duck Developer
"""

import os
import sys
import json
import shutil
import subprocess
from pathlib import Path
from datetime import datetime
from typing import Optional, List

# Color output for terminal
class Colors:
    GREEN = '\033[92m'
    YELLOW = '\033[93m'
    RED = '\033[91m'
    BLUE = '\033[94m'
    RESET = '\033[0m'
    DUCK = '🦆'

def print_header(msg: str):
    print(f"{Colors.BLUE}{Colors.DUCK} {msg}{Colors.RESET}")

def print_success(msg: str):
    print(f"{Colors.GREEN}✓ {msg}{Colors.RESET}")

def print_warning(msg: str):
    print(f"{Colors.YELLOW}[!] {msg}{Colors.RESET}")

def print_error(msg: str):
    print(f"{Colors.RED}✗ {msg}{Colors.RESET}")

class ProjectScaffolder:
    """Pure Python project scaffolder with Git security hooks."""
    
    def __init__(self, project_name: str, base_path: Optional[str] = None):
        self.project_name = project_name
        self.base_path = Path(base_path or os.getcwd())
        self.project_path = self.base_path / project_name
        self.git_hooks_dir = self.project_path / ".git" / "hooks"
        
    def validate_project_name(self) -> bool:
        """Ensure project name is valid and doesn't already exist."""
        if self.project_path.exists():
            print_error(f"Project '{self.project_name}' already exists!")
            return False
        
        if not self.project_name.replace('_', '').replace('-', '').isalnum():
            print_error("Project name must be alphanumeric (hyphens/underscores OK)")
            return False
        
        return True
    
    def create_directory_structure(self):
        """Build the project directory tree."""
        print_header(f"Creating project structure for '{self.project_name}'...")
        
        directories = [
            self.project_path / "src",
            self.project_path / "tests",
            self.project_path / "docs",
            self.project_path / "scripts",
            self.project_path / ".github" / "workflows",
        ]
        
        for directory in directories:
            directory.mkdir(parents=True, exist_ok=True)
            print_success(f"Created {directory.relative_to(self.base_path)}")
    
    def create_claudeignore(self):
        """Create .claudeignore for token optimization."""
        claudeignore_content = """# Dependencies
node_modules/
venv/
.venv/
env/
__pycache__/
*.pyc
*.pyo
*.egg-info/

# OS/IDE
.git/
.vscode/
.idea/
.DS_Store
*.swp
*.swo
*~

# Build artifacts
dist/
build/
*.exe
*.out
*.o

# Large data
*.csv
*.jsonl
*.sql
*.db
data/
datasets/

# Logs
*.log
logs/

# Environment
.env
.env.local
secrets/
"""
        claudeignore_path = self.project_path / ".claudeignore"
        claudeignore_path.write_text(claudeignore_content)
        print_success("Created .claudeignore")
    
    def create_readme(self):
        """Generate project README."""
        readme_content = f"""# {self.project_name}

> An AI-ready project scaffolded by **hatch.py** on {datetime.now().strftime('%Y-%m-%d %H:%M')}

## Quick Start

```bash
cd {self.project_name}
python3 -m venv venv
source venv/bin/activate  # On Windows: venv\\Scripts\\activate
pip install -r requirements.txt
```

## Project Structure

```
{self.project_name}/
├── src/              # Main source code
├── tests/            # Unit and integration tests
├── docs/             # Documentation
├── scripts/          # Utility scripts
├── .github/          # GitHub workflows (CI/CD)
└── .claudeignore     # Token-aware file exclusions
```

## Security

This project includes automated Git hooks that prevent:
- 🚫 Committing API keys or secrets
- 🚫 Trailing whitespace pollution
- 🚫 Broken Python syntax

These hooks run automatically before every commit.

## Requirements

See `requirements.txt` for pinned dependencies.

---

*Generated by hatch.py — Quack!*
"""
        readme_path = self.project_path / "README.md"
        readme_path.write_text(readme_content)
        print_success("Created README.md")
    
    def create_requirements(self):
        """Generate requirements.txt with sensible defaults."""
        requirements_content = """# Core dependencies
requests>=2.31.0
python-dotenv>=1.0.0

# Development & Testing
pytest>=7.4.0
pytest-cov>=4.1.0
black>=23.0.0
flake8>=6.0.0
isort>=5.12.0

# Async & Performance
aiohttp>=3.8.0
httpx>=0.24.0

# Data & Logging
pydantic>=2.0.0
"""
        req_path = self.project_path / "requirements.txt"
        req_path.write_text(requirements_content)
        print_success("Created requirements.txt")
    
    def create_git_hooks(self):
        """Set up automated Git security hooks."""
        print_header("Wiring up Git security hooks...")
        
        # Initialize git repo if not already done
        git_dir = self.project_path / ".git"
        if not git_dir.exists():
            subprocess.run(
                ["git", "init"],
                cwd=self.project_path,
                capture_output=True
            )
            print_success("Initialized Git repository")
        
        self.git_hooks_dir.mkdir(parents=True, exist_ok=True)
        
        # Pre-commit hook
        pre_commit_hook = self._create_pre_commit_hook()
        hook_path = self.git_hooks_dir / "pre-commit"
        hook_path.write_text(pre_commit_hook)
        hook_path.chmod(0o755)
        print_success("Installed pre-commit hook")
        
        # Commit message template
        commit_template = self._create_commit_template()
        template_path = self.project_path / ".gitmessage"
        template_path.write_text(commit_template)
        
        # Configure git to use the template
        subprocess.run(
            ["git", "config", "commit.template", ".gitmessage"],
            cwd=self.project_path,
            capture_output=True
        )
        print_success("Installed commit message template")
    
    def _create_pre_commit_hook(self) -> str:
        """Generate the pre-commit hook script."""
        return f"""#!/bin/bash
# Pre-commit hook: Guard against secrets, syntax errors, and whitespace pollution
# Generated by hatch.py

set -e

PROJECT_ROOT="$(git rev-parse --show-toplevel)"
STAGED_FILES=$(git diff --cached --name-only)

echo "🦆 Running pre-commit checks..."

# Check 1: Prevent accidental secret commits
echo "[*] Checking for secrets..."
if git diff --cached | grep -E '(ANTHROPIC_API_KEY|github_token|password|secret|DATABASE_URL)' > /dev/null; then
    echo "❌ QUACK! Secret detected in staged changes!"
    echo "   Remove the secret and try again:"
    echo "   git reset HEAD <file>"
    exit 1
fi

# Check 2: Python syntax validation
echo "[*] Validating Python syntax..."
for py_file in $(echo "$STAGED_FILES" | grep '\\.py$'); do
    if [ -f "$py_file" ]; then
        python3 -m py_compile "$py_file" || exit 1
    fi
done
echo "   ✓ Python syntax OK"

# Check 3: Trailing whitespace check
echo "[*] Checking for trailing whitespace..."
if git diff --cached | grep -E '^\\+.*[[:space:]]$' > /dev/null; then
    echo "⚠️  Trailing whitespace detected!"
    echo "   Run: git diff --cached | grep -E '^\\+.*[[:space:]]$'"
    exit 1
fi
echo "   ✓ No trailing whitespace"

echo "✓ All checks passed! Quack! 🦆"
exit 0
"""
    
    def _create_commit_template(self) -> str:
        """Generate the commit message template."""
        return """# <type>(<scope>): <subject>
# 
# Example: feat(auth): add API key validation
#
# Types: feat, fix, docs, style, refactor, perf, test, chore, ci
# Scope: the component or module affected
# Subject: keep under 50 characters, use imperative mood
# 
# <body>
# Explain WHAT and WHY, not HOW (the code shows HOW)
# Wrap at 72 characters.
# 
# <footer>
# Closes #123
# BREAKING CHANGE: description if applicable
#
# --- Generated by hatch.py ---
"""
    
    def create_github_workflow(self):
        """Create a basic GitHub Actions CI/CD workflow."""
        workflow_content = f"""name: CI Pipeline

on:
  push:
    branches: [ main, develop ]
  pull_request:
    branches: [ main ]

jobs:
  test:
    runs-on: ubuntu-latest
    strategy:
      matrix:
        python-version: ['3.9', '3.10', '3.11', '3.12']
    
    steps:
      - uses: actions/checkout@v3
      
      - name: Set up Python ${{{{ matrix.python-version }}}}
        uses: actions/setup-python@v4
        with:
          python-version: ${{{{ matrix.python-version }}}}
      
      - name: Install dependencies
        run: |
          python -m pip install --upgrade pip
          pip install -r requirements.txt
      
      - name: Lint with flake8
        run: flake8 src/ tests/ --count --select=E9,F63,F7,F82 --show-source --statistics
      
      - name: Run tests
        run: pytest tests/ -v --cov=src/
"""
        workflow_path = self.project_path / ".github" / "workflows" / "ci.yml"
        workflow_path.write_text(workflow_content)
        print_success("Created GitHub Actions CI workflow")
    
    def create_initial_commit(self):
        """Scaffold the project with an initial Git commit."""
        print_header("Creating initial commit...")
        
        try:
            subprocess.run(
                ["git", "add", "."],
                cwd=self.project_path,
                capture_output=True
            )
            subprocess.run(
                ["git", "commit", "-m", f"Initial scaffold: {self.project_name}"],
                cwd=self.project_path,
                capture_output=True
            )
            print_success("Initial commit created")
        except Exception as e:
            print_warning(f"Could not create initial commit: {e}")
    
    def scaffold(self):
        """Execute the full scaffolding pipeline."""
        if not self.validate_project_name():
            return False
        
        try:
            self.create_directory_structure()
            self.create_claudeignore()
            self.create_readme()
            self.create_requirements()
            self.create_git_hooks()
            self.create_github_workflow()
            self.create_initial_commit()
            
            print("\n" + "="*60)
            print_success(f"Project '{self.project_name}' is ready to dive!")
            print("="*60)
            print(f"\n{Colors.DUCK} Next steps:\n")
            print(f"  cd {self.project_name}")
            print(f"  python3 -m venv venv")
            print(f"  source venv/bin/activate")
            print(f"  pip install -r requirements.txt")
            print(f"\n  Then launch Claude:\n")
            print(f"  claude 'Help me implement X'\n")
            print(f"Quack! 🦆\n")
            
            return True
        except Exception as e:
            print_error(f"Scaffolding failed: {e}")
            return False


def main():
    """CLI entry point."""
    if len(sys.argv) < 2:
        print(f"{Colors.DUCK} hatch.py - AI-Ready Project Bootstrapper\n")
        print("Usage: python3 hatch.py <project_name> [base_path]\n")
        print("Example:")
        print("  python3 hatch.py my_ai_project")
        print("  python3 hatch.py another_project ~/projects\n")
        sys.exit(1)
    
    project_name = sys.argv[1]
    base_path = sys.argv[2] if len(sys.argv) > 2 else None
    
    scaffolder = ProjectScaffolder(project_name, base_path)
    success = scaffolder.scaffold()
    
    sys.exit(0 if success else 1)


if __name__ == "__main__":
    main()
